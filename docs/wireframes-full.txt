// Wireframe primitives — grayscale, tight to FadeJunkie type scale.
// Real type system, fake visuals. Dashed = placeholder imagery, solid = real component.

const WIRE = {
  ink: 'rgba(0,0,0,0.88)',
  ink2: 'rgba(0,0,0,0.6)',
  ink3: 'rgba(0,0,0,0.38)',
  line: 'rgba(0,0,0,0.12)',
  lineHard: 'rgba(0,0,0,0.3)',
  bg: '#ffffff',
  bgAlt: '#f4f3f1',
  bgDark: '#31302e',
  fill: 'rgba(0,0,0,0.04)',
  fill2: 'rgba(0,0,0,0.08)',
  accent: '#0075de',
  accentSoft: '#e8f2fc',
  note: '#fef4a8',
  noteInk: '#5a4a2a',
};

// Base fills a square region with diagonal stripes — our "placeholder image" glyph
function Placeholder({ label, style = {}, ratio, h }) {
  return (
    <div style={{
      position: 'relative',
      width: '100%',
      aspectRatio: ratio,
      height: h,
      background: `repeating-linear-gradient(135deg, ${WIRE.fill} 0 6px, transparent 6px 12px)`,
      border: `1px dashed ${WIRE.lineHard}`,
      borderRadius: 4,
      display: 'flex',
      alignItems: 'center',
      justifyContent: 'center',
      color: WIRE.ink3,
      fontFamily: 'ui-monospace, SFMono-Regular, Menlo, monospace',
      fontSize: 10,
      letterSpacing: 0.3,
      textTransform: 'uppercase',
      ...style,
    }}>{label}</div>
  );
}

// Solid grayscale block for "text block" wireframe lines
function Lines({ count = 3, width = '100%', lastWidth = '70%', gap = 6, color = WIRE.line, h = 6 }) {
  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap }}>
      {Array.from({ length: count }).map((_, i) => (
        <div key={i} style={{
          height: h,
          width: i === count - 1 ? lastWidth : width,
          background: color,
          borderRadius: 2,
        }} />
      ))}
    </div>
  );
}

// A single solid bar representing a heading
function Bar({ w = '60%', h = 14, color = WIRE.ink, style = {} }) {
  return <div style={{ height: h, width: w, background: color, borderRadius: 3, ...style }} />;
}

// A pill badge
function Pill({ children, color = WIRE.accent, bg = WIRE.accentSoft }) {
  return (
    <span style={{
      display: 'inline-flex', alignItems: 'center', gap: 4,
      fontSize: 10, fontWeight: 600, letterSpacing: 0.5, textTransform: 'uppercase',
      color, background: bg, padding: '3px 8px', borderRadius: 9999, lineHeight: 1.2,
    }}>{children}</span>
  );
}

// A button
function Btn({ children, primary, ghost, full, size = 'md' }) {
  const pad = size === 'lg' ? '10px 18px' : size === 'sm' ? '5px 10px' : '8px 14px';
  const fs = size === 'lg' ? 14 : size === 'sm' ? 11 : 12;
  return (
    <div style={{
      display: 'inline-flex', alignItems: 'center', justifyContent: 'center', gap: 6,
      padding: pad, fontSize: fs, fontWeight: 600, borderRadius: 4,
      background: primary ? WIRE.accent : ghost ? 'transparent' : WIRE.fill,
      color: primary ? '#fff' : WIRE.ink,
      border: ghost ? `1px solid ${WIRE.line}` : 'none',
      width: full ? '100%' : 'auto',
      whiteSpace: 'nowrap',
    }}>{children}</div>
  );
}

// Card wrapper
function Card({ children, pad = 16, style = {}, alt, dashed }) {
  return (
    <div style={{
      background: alt ? WIRE.bgAlt : '#fff',
      border: dashed ? `1px dashed ${WIRE.lineHard}` : `1px solid ${WIRE.line}`,
      borderRadius: 8,
      padding: pad,
      boxShadow: dashed ? 'none' : '0 1px 2px rgba(0,0,0,0.04), 0 2px 8px rgba(0,0,0,0.03)',
      ...style,
    }}>{children}</div>
  );
}

// Margin annotation — shown when annotations toggle is on
function Note({ children, side = 'right', top, width = 140 }) {
  const isRight = side === 'right';
  return (
    <div data-wire-note style={{
      position: 'absolute',
      top,
      [isRight ? 'left' : 'right']: '100%',
      [isRight ? 'marginLeft' : 'marginRight']: 16,
      width,
      padding: '8px 10px',
      background: WIRE.note,
      color: WIRE.noteInk,
      fontFamily: '"Caveat", "Marker Felt", cursive',
      fontSize: 14,
      lineHeight: 1.25,
      boxShadow: '0 1px 3px rgba(0,0,0,0.12)',
      transform: `rotate(${isRight ? -1.5 : 1.5}deg)`,
      pointerEvents: 'none',
      zIndex: 3,
    }}>
      <div style={{
        position: 'absolute',
        top: 14,
        [isRight ? 'left' : 'right']: -16,
        width: 16, height: 1,
        borderTop: `1px dashed ${WIRE.noteInk}`,
        opacity: 0.5,
      }} />
      {children}
    </div>
  );
}

// Device frame — gives wireframes a consistent canvas at mobile OR desktop scale
function Frame({ children, w, h, label }) {
  return (
    <div style={{
      width: w, height: h, background: '#fff',
      border: `1px solid ${WIRE.line}`,
      position: 'relative', overflow: 'hidden',
      fontFamily: "'Inter', -apple-system, system-ui, sans-serif",
      color: WIRE.ink,
    }}>
      <div style={{ height: '100%', overflowY: 'auto' }}>
        {children}
      </div>
    </div>
  );
}

// Full-bleed scroll container for a page
function Page({ children, dark, alt }) {
  return <div style={{ background: dark ? WIRE.bgDark : alt ? WIRE.bgAlt : '#fff', minHeight: '100%', color: dark ? '#fff' : WIRE.ink }}>{children}</div>;
}

    <div style={{

    }}>

              : 'Most effective way to sanitize metal implements between clients?'}
          </div>
        </div>
      </div>
      {/* Footer */}
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', marginTop: 10, fontSize: 11, color: WIRE.ink2 }}>
        <span>{i} / 300</span>
        <div style={{ display: 'flex', gap: 6 }}>

                      cursor: r.soon ? 'default' : 'pointer',
                      boxShadow: isOpen ? `0 0 0 3px ${WIRE.accentSoft}` : 'none',
                      position: 'relative',
                    }}
                  >
                    <img src={r.icon} alt="" style={{ width: 56, height: 56, objectFit: 'contain', flexShrink: 0 }} />

                      <div style={{ fontSize: 13, color: WIRE.ink2, marginTop: 4 }}>{r.s}</div>
                      {!r.soon && (
                        <div style={{ marginTop: 10, fontSize: 11, textTransform: 'uppercase', letterSpacing: 0.8, fontWeight: 700, color: isOpen ? WIRE.accent : WIRE.ink3 }}>
                          {isOpen ? '← showing preview' : 'Try inline →'}
                        </div>
                      )}
                    </div>
                    {r.soon && <span style={{ position: 'absolute', top: 16, right: 16, fontSize: 10, textTransform: 'uppercase', color: WIRE.ink3 }}>{r.soon}</span>}
                  </div>
                );
              })}
            </div>

            {/* Widget panel */}
            <div style={{ position: 'sticky', top: 88 }}>
              {Embed ? (
                <Embed onMin={() => setOpen(null)} />
              ) : (
                <div style={{ background: WIRE.bgAlt, border: `1.5px dashed ${WIRE.line}`, borderRadius: 12, padding: 40, textAlign: 'center' }}>
                  <div style={{ fontSize: 14, color: WIRE.ink2 }}>Pick a mode on the left to preview it here →</div>
                </div>
              )}
              {annot && <Note top={20} side="right" width={170}>desktop: rows + sticky widget panel. rows never collapse. minimize returns to the empty-state placeholder.</Note>}
            </div>
          </div>
        </div>
      </Sec>

      <Footer />
    </Page>
  );
